ཡིག་ཨང་། ཆོས་༦།ནང་ཆོས་ངོ་སྤྲོད།༢༠༡༤་་་་༡༥།

ཟླ་དྲུག་ཟབ་ཁྲིད་ཐེངས་གཉིས་པའི་གསལ་བསྒྲགས།

༄༅།། བོད་མིའི་སྒྲིག་འཛུགས་ཆོས་རིག་ལས་ཁུངས་ནས་༸གོང་ས་༸སྐྱབས་མགོན་ཆེན་པོའི་རླབས་ཆེན་བཀའ་སློབ་དགོངས་བཞེད་སྒྲུབ་ཐབས་དང་། ལྷག་པར་བོད་ཀྱི་ཆོས་དང་རིག་གཞུང་མི་ཉམས་རྒྱུན་འཛིན་ལ་དགོངས་ཏེ་ན་ནིང་ལོར་ཧུན་སུར་གཞིས་ཆགས་སུ་རྟེན་གཞི་བྱེད་པའི་རྒྱུད་སྨད་གངས་ཅན་རིག་གནས་སློབ་གྲྭ་དང་ཟུང་འབྲེལ་ཐོག བོད་རིགས་གཞོན་སྐྱེས་མཐོ་སློབ་མཐར་སོན་བཅུ་ཕྲག་གཉིས་ལ་བོད་ཀྱི་སྐད་ཡིག་དང་ནང་དོན་རིག་པའི་ཟབ་ཁྲིད་ཅིག་གོ་སྒྲིག་གནང་བ་དེ་ཉིད། དོན་ཕན་དགེ་མཚན་ཆེ་ལུགས་དང་མུ་མཐུད་ལག་བསྟར་དགོས་པའི་དགོངས་ཚུལ་ཡང་ཡང་འབྱུང་བཞིན་པ་གཞིར་བཅས། ༸གོང་ས་མཆོག་གི་དགོངས་བཞེད་ལྟར་སྐྱ་བོ་ཚོས་ཀྱང་སེར་མོ་བ་དང་གཅིག་མཚུངས་སྤྱིར་བོད་ཀྱི་སྐད་ཡིག་དང་། ཡང་སྒོས་དབུ་ཕར་སོགས་ནང་ཆོས་ཀྱི་རིག་གནས་ཁག་ལ་ཚད་མཐོའི་སློབ་གཉེར་གནང་ཐུབ་པའི་དམིགས་ཡུལ་འོག་ ༢༠༡༥ ལོར་བོད་ཀྱི་སྐད་ཡིག་དང་ནང་དོན་རིག་པའི་ཟབ་ཁྲིད་ཐེངས་གཉིས་པ་འཚུག་གཏན་འཁེལ་བར། ལས་འཆར་དེའི་ཁོངས་སུ་ཞུགས་འདོད་མཐོ་སློབ་མཐར་སོན་རང་རིགས་སློབ་ཕྲུག་རྣམས་ནས་གཤམ་གསལ་དོན་ཚན་ཁག་ལ་ཐུགས་ཞིབ་དང་འབྲེལ། བོད་མིའི་སྒྲིག་འཛུགས་དང་། ཆོས་རིག་ལས་ཁུངས། ཤེས་རིག་ལས་ཁུངས་བཅས་ཀྱི་དྲ་རྒྱ་གང་རུང་ནས་འགེངས་ཤོག་ཕབ་ལེན་རྗེས་འཐུས་གཙང་བཀང་སྟེ་ཆ་རྐྱེན་ཡིག་ཆ་ལྷན་དུ་དུས་ཚིགས་ནང་ཚུད་ཆོས་རིག་ལས་ཁུངས་སུ་ངེས་འབུལ་དགོས་པའི་གསལ་བསྒྲགས་སུ།

  1. འཚུག་ཡུལ།

རྒྱ་གར་ལྷོ་ཕྱོགས་ཧུན་སུར་དཔལ་ལྡན་རྒྱུད་སྨད་གྲྭ་ཚང་།

  1. ཟབ་ཁྲིད་དུས་ཡུན།

ཕྱི་ལོ་ ༢༠༡༥ ཟླ་ ༧ ཚེས་ ༡ ཟླ་ ༡༢ ཚེས་ ༣༡ བར་ཟླ་བ་དྲུག

  1. ལམ་གྲོན་དང་བཞུགས་གནས་སོགས་ཀྱི་མཐུན་རྐྱེན།

ཕར་ཚུར་ལམ་གྲོན་དང་ས་གནས་སུ་བཞུགས་གནས་བཞེས་ལག་བཅས་ཀྱི་མཐུན་རྐྱེན་ཆོས་རིག་ལས་ཁུངས་དང་རྒྱུད་སྨད་གངས་ཅན་སློབ་གྲྭ་ནས་གནང་རྒྱུ།

  1. འཛུལ་ཞུགས་པའི་དགོས་ངེས་ཆ་རྐྱེན།

ཀ༽ དེང་དུས་མཐོ་སློབ་ཐོན་པ་དང་།

ཁ༽ འཛིན་རིམ་བཅུ་གཉིས་བར་བོད་ཡིག་སྦྱངས་མྱོང་བ། ཡང་ན་བོད་ཡིག་ཤེས་ཚད་ལྡེང་ངེས་ཡོད་པ་དང་།

ག༽ ལས་རིམ་འདིའི་ཁོངས་སུ་འཛུལ་ཞུགས་རྗེས་སློབ་དུས་མ་རྫོགས་བར་ཀར་ཀོར་མེད་པར་ས་གནས་སུ་

བཞུགས་ཐུབ་མཁན་ཡིན་པ་བཅས་ཡིན་དགོས།

  1. སློབ་གཉེར་བའི་ས་མིག

ཁྱོན་སློབ་གཉེར་བའི་ས་མིག་སོ་གཉིས་ (༣༢) ཡིན། དེ་ཉིད་སྙན་ཞུ་འབྱོར་སྔ་རིམ་བཞིན་འདེམས་རྒྱུ།

  1. སློབ་ཚན།

ཀ༽ སྐད་ཡིག

བརྡ་སྤྲོད།

༡ དབྱངས་ཅན་གྲུབ་པའི་རྡོ་རྗེས་མཛད་པའི་སུམ་ཅུ་པའི་སྙིང་པོ་ལེགས་བཤད་ལྗོན་དབང་གི་འགྲེལ་པ་དང་། ༢ དཀའ་ཆེན་པདྨས་མཛད་པའི་རྟགས་འཇུག་དཀའ་གནད་གསལ་བའི་མེ་ལོང་གི་འགྲེལ་པ་རིགས་ལམ་གསེར་གྱི་ལྡེ་མིག

སྙན་ངག

༡ སྤྱི་བཤད། ༢ མེ་ལོང་མའི་བསྡུས་དོན། ༣ སྙན་ངག་གི་ལུས། ༤ སྙན་ངག་གི་རྒྱན། ༥ རང་བཞིན་བརྗོད་པ། ༦ དཔེ་རྒྱན།

དག་ཡིག

༡ བྱིས་པ་དག་ཡིག ༢ བོད་ཡིག་གི་དུས་གསུམ་རྣམ་གཞག

འབྲི་རྩོམ།

༡ རྩོམ་རིག ༢ ཡིག་བསྐུར་རྣམ་གཞག

ཁ༽ ནང་དོན་རིག་པ།

སྔོན་འགྲོ

ཆོས་འབྱུང་།

༡ སྟོན་པའི་མཛད་རྣམ་སྙིང་བསྡུས། ༢ ཆོས་འཁོར་གསུམ། ༣ བཀའ་བསྡུ་རིམ་པ་གསུམ། ༤ བཀའ་བསྟན་གྱི་རྣམ་གཞག།༥ སྡེ་སྣོད་གསུམ་བྱེ་བྲག་ཏུ་བཤད་པ། ༦ བོད་ཀྱི་ཆོས་བརྒྱུད་ཁག་གི་ངོ་སྤྲོད་མདོར་བསྡུས།

གྲུབ་མཐའ།

༡ ཕྱི་རོལ་པའི་གྲུབ་མཐའ་མདོར་བསྡུས་བཤད་པ། ༢ རང་སྡེའི་གྲུབ་མཐའ་བཞི་རྒྱས་པར་བཤད་པ།

བསྡུས་གྲྭ

༡ སྤྱི་བཤད། ༢ རིགས་ལམ་སྦྱོང་བའི་ཚུལ། ༣ མུ་གསུམ་མུ་བཞི། ༤ ཁྱབ་པ་སྒོ་བརྒྱད། ༥ མཚན་མཚོན་གཞི་གསུམ། ༦ ཤེས་བྱ་ཡུལ་གྱི་རྣམ་གཞག ༧ རྒྱུ་དང་འབྲས་བུའི་རྣམ་གཞག ༨ གཅིག་དང་ཐ་དད། ༩ སྤྱི་དང་བྱེ་བྲག ༡༠ འགལ་འབྲེལ་རྣམ་གཞག ༡༡ དགག་སྒྲུབ་ཀྱི་རྣམ་གཞག

བློ་རིག

༡ སྤྱི་བཤད། ༢ ཡུལ་ཅན་བློའི་དབྱེ་ཚུལ། ༣ བླ་རིག་གི་དབྱེ་བ་བྱེ་བྲག་ཏུ་བཤད་པ། ༤ སེམས་སེམས་བྱུང་གི་རྣམ་གཞག་རྒྱས་པར་བཤད་པ།

རྟགས་རིགས།

༡ གཏན་ཚིགས་ཀྱི་རྣམ་གཞག ༢ རིགས་པ་བཞིའི་རྣམ་གཞག ༣ ཐལ་འགྱུར་གྱི་རྣམ་གཞག ༤ སྒྲུབ་ངག་གི་རྣམ་གཞག ༥ གཏན་ཚིགས་ཡང་དག་དང་ལྟར་སྣང་གཉིས་བཤད་པ།

གཞུང།

ཕར་ཕྱིན།

༡ སྤྱི་བཤད། ༢ སྐྱབས་གསུམ་གྱི་རྣམ་གཞག ༣ བདེན་པ་བཞི། ༤ ས་ལམ་གྱི་རྣམ་གཞག

ཚད་མ།

༡ སྤྱི་བཤད། ༢ སྟོན་པ་ཚད་མར་སྒྲུབ་པའི་རིགས་པ། ༣ སྐྱེ་བ་སྔ་ཕྱི་སྒྲུབ་པའི་རིགས་པ། ༤ ཐར་པ་དང་ཐམས་ཅད་མཁྱེན་པ་སྒྲུབ་པའི་རིགས་པ།

དབུ་མ།

༡ སྤྱི་བཤད། ༢ རྟེན་འབྲེལ་ཡན་ལག་བཅུ་གཉིས། ༣ འཁོར་བ་ལྡོག་ཚུལ། ༤ དེ་ཁོ་ན་ཉིད་ཀྱི་ལྟ་བ་གཏན་ལ་དབབ་པ། ༥ སྟོང་པ་རྟེན་འབྱུང་གི་གོ་དོན།

མཛོད།

༡ སྤྱི་བཤད། ༢ ཕུང་ཁམས་སྐྱེ་མཆེད་བཤད་པ། ༣ མཛོད་གནས་གསུམ་པའི་དོན་མདོ་ཙམ་བཤད་པ།

སྔགས།

༡ སྤྱི་བཤད། ༢ མདོ་སྔགས་གཉིས་ཀྱི་ཁྱད་པར། ༣ སྔགས་ས་ལམ་ཅུང་ཟད་ཕྱེ་སྟེ་བཤད་པ།

  1. མཁས་དབང་གསུང་བཤད་དང་ཤེས་ཡོན་བལྟ་བསྐོར།

ཟླ་བ་རེ་རེའི་ནང་རིས་སུ་མ་ཆད་པའི་མཁས་དབང་རེ་གདན་ཞུས་ཀྱིས་དམིགས་བསལ་གསུང་བཤད་དང་། སློབ་ཡུན་བར་མཚམས་སུ་སྦེལ་ཀོབ་ཁུལ་གྱི་དགོན་སྡེ་གནས་གཟིགས་སོགས་ཤེས་ཡོན་བལྟ་བསྐོར་ཐེངས་གཉིས་འཁྲིད་རྒྱུ།

  1. རྒྱུགས་སྤྲོད་དང་ལག་འཁྱེར།

སློབ་འཁྲིད་ཟིན་ལ་ཉེ་སྐབས་རྩོད་རྒྱུགས་དང་འབྲི་རྒྱུགས་བཅས་བཏང་སྟེ་གྲུབ་འབྲས་ཨང་ཤོག་དང་ཟབ་ཁྲིད་མཐར་སོོན་གྱི་ལག་ཁྱེར་ཞིག་སྤྲོད་རྒྱུ།

  1. སྙན་ཞུ་འབུལ་ཚེས་མཐའ་མ།

ཕྱི་ལོ་ ༢༠༡༥ ཟླ་ ༥ ཚེས་ ༡༥།

  1. འགེངས་ཤོག་མཉམ་དུ་དགོས་ངེས་ཡིག་ཆ།

ཀ༽ མཐོ་སློབ་མཐར་སོན་ལག་འཁྱེར་གྱི་ངོ་བཤུས་དང་།

ཁ༽ ༢༠༡༤ བར་འབུལ་འབབ་གཙང་བའི་དྭང་བླང་དཔྱ་དངུལ་ལག་དེབ་ཀྱི་ངོ་བཤུས།

ག༽ ཉེ་ཆར་ཕྱེད་པར་གཅིག་བཅས།

  1. སྙན་ཞུ་འབུལ་ཡུལ་དང་འབྲེལ་བ་བྱ་ཡུལ།
